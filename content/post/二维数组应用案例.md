# 1.使用二维数组打印一个10行杨辉三角
```#
    public static void main(String[] args) {
        int[][] arr = new int[10][];
        for (int i = 0; i < arr.length; i++) {
            //给一维数组的子数组开辟空间
            arr[i] = new int[i+1];
            for (int j = 0; j < arr[i].length; j++) {
                if (j==0 || j== arr[i].length-1){
                    arr[i][j] = 1;
                }else{
                    arr[i][j] = arr[i-1][j] + arr[i-1][j-1];
                }
                System.out.print(arr[i][j]+ "\t");
            }
            System.out.println(" ");
        }
    }

# 3 随机生成10个整数保存到数组 并倒序打印，求平均值，求最大值及下标，并查找是否有8
```
public static void main(String[] args) {
        int[] arr = new int[10];
        int sum = 0;
        int max = 0;
        int maxindex = 0;
        boolean flag = false;
        for (int i = 0; i < arr.length; i++) {
            int rom = (int) (Math.random()*100) +1;
            arr[i] = rom;
            System.out.print(arr[i]+"\t");
        }
        System.out.print("\n下面是倒序******************\n");
        for (int i = arr.length-1; i >=0; i--) {
            System.out.print(arr[i]+"\t");
            sum += arr[i];
            if (max < arr[i]){
                max = arr[i];
                maxindex = i;
            }
            if (arr[i] == 8){
                flag = true;
            }
        }
        System.out.println("\n平均值是"+ sum/ arr.length);
        System.out.println("\nmax = "+max + "下标[" + maxindex + "]");
        if (flag){
            System.out.println("有8");
        }else {
            System.out.println("没8");
        }
    }
